3. Find the sum and average of elements in an array

#include <stdio.h>

int main() {
    int arr[6], i;
    int sum = 0;
    float average;

    printf("Enter 6 elements:\n");
    for(i = 0; i < 6; i++) {
        scanf("%d", &arr[i]);
        sum += arr[i];
    }

    average = (float)sum / 6;

    printf("Sum of elements: %d\n", sum);
    printf("Average of elements: %.2f\n", average);

    return 0;
}
